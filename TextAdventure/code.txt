import java.util.*;import java.io.*;import java.awt.event.*;
public class TextEngine implements KeyListener{    static String location = "Menu";    static Scanner userInput = new Scanner(System.in);    static volatile boolean continuing = false;    static String lastDecision;    public static void main(String []args){        System.out.println("Hello player.");        System.out.println("You are entering a world that is free of the laws that burden our natural world.");        System.out.println("While the majority of the population is happy for the possibilities that this world provides, some are not so satisfied.");        System.out.println("Taking advantage of the endless paths to destruction, these villains claim to be invincible. It seems like you typed something wrong. Please try again.");                    }                }            }            if(location.equals("Game Start")){                System.out.println("Welcome to Cocalmairell. It's a magical place.");            }        }    }    public static void spaceToContinue(){        System.out.println("Press SPACE to continue.");        while(!continuing){                    }    }    public void keyPressed(KeyEvent e){        if(e.getKeyCode()==KeyEvent.VK_SPACE){            continuing = true;        }    }    public void keyReleased(KeyEvent e){            }    public void keyTyped(KeyEvent e){            }    }
